import heapq
#Define the graph as an adjacency list
graph = {
 'A': ['B', 'C', 'D'],
 'B': ['E'],
 'C': [],
 'D': ['G'],
 'E': [],
 'G': []
}
#Heuristic values for each node
heuristic = {
'A': 5,
'B': 4,
'C': 3,
'D': 2,
'E': 6,
'G': 0
}
def best_first_search(start, goal):
 visited = set()   # visited set
 Piority_queue = []
 heapq.heappush(Piority_queue, (heuristic[start], start)) # heuristic of A (5) , A  ..(5,A)
 while Piority_queue:
        # h will receive the priority of the node, and node will receive the name of the node
   h, node = heapq.heappop(Piority_queue)   # h is 5 and  node is A, next h is 2 and node is D
   if node in visited:                                          # next h is 0 and node G
    continue
   print(f"Visited: {node}")
   visited.add(node)     # visited  A,  D visited  G visited   
   if node == goal:   # no,  no   yes
    print("Goal reached!")
    return
   for neighbor in graph[node]:  # graph [A]    neighbor B,C,D, neighbor G 
    if neighbor not in visited:  # B   G
     heapq.heappush(Piority_queue, (heuristic[neighbor], neighbor)) #pushed( B:4,C:3,D:2) 
     #Run the algorithm                            #pushed (G:0) the queue is (G:0, B:4,C:3)
best_first_search('A', 'G')
